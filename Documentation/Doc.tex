		\item TPS259571DSGR -- Siguranta electronica (eFuse) cu monitorizare si protectie

		\item FT230XQ -- Circuit interfata punte USB la UART serial

	\section{Specificații finale \\}
	
	\begin{itemize}
		\item Backwards compatibility cu Arduino IDE (maparea pinilor este realizată în stil Arduino Mega)
		\item Frecvență maximă a semnalului de măsurat circa 300kHz (în funcție de configurarea ADC-ului, limită impusă de amplificatorul de intrare)
		\item Alimentare unică prin USB-C de 5V
		\item Transfer de date prin USB-C
		\item Intrare BNC
		\item Cuplaj AC/DC
		\item Encodere digitale pentru control al afișajului (baza de timp și scară de tensiune)
		\item Tensiune semnal de intrare >230Vac
		\item Dublă izolare galvanică pe interfața de intrare a semnalelor.
		
	\end{itemize}
